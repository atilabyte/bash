elf=0

out=$(  file  file.elf )


for  f  in $out ; do   

 if  [[ $f = 'ELF' ]] ;   then
   
   elf=1

      fi;
done 

# elf is   1 or 0

#042577

octal=$(   od file.elf  | head -n  1 )
 
for   oc in $octal ; do


    if [ $oc -eq  042577 ] ; then 
     
       elf=$(( $elf + 1 )) #more verification


        fi;

done
